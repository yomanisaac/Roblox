local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

local waypointCFrame = nil
local locked = false

local SET_WAYPOINT_KEY = Enum.KeyCode.F5
local TOGGLE_KEY = Enum.KeyCode.F6

local function getHRP()
	local character = player.Character
	if not character then return nil end
	return character:FindFirstChild("HumanoidRootPart")
end

local coordBox, lockButton

local function updateLockButton()
	if not lockButton then return end
	lockButton.Text = locked and "Lock: ON" or "Lock: OFF"
	lockButton.TextColor3 = locked and Color3.fromRGB(120, 220, 140) or Color3.fromRGB(220, 220, 220)
end

local function parseCoords(text)
	local nums = {}
	for n in text:gmatch("-?%d+%.?%d*") do
		table.insert(nums, tonumber(n))
	end
	if #nums >= 3 then
		return Vector3.new(nums[1], nums[2], nums[3])
	end
	return nil
end

local function setWaypoint()
	local typed = coordBox and parseCoords(coordBox.Text) or nil
	if typed then
		waypointCFrame = CFrame.new(typed)
	else
		local hrp = getHRP()
		if hrp then
			waypointCFrame = hrp.CFrame
		else
			warn("[Isaac's Lock] Could not find character.")
			return
		end
	end
	if coordBox then
		local p = waypointCFrame.Position
		coordBox.Text = string.format("%.1f, %.1f, %.1f", p.X, p.Y, p.Z)
	end
end

local function toggleLock()
	if waypointCFrame == nil then
		warn("[Isaac's Lock] No waypoint set yet.")
		return
	end
	locked = not locked
	updateLockButton()
end

UserInputService.InputBegan:Connect(function(input, gameProcessedEvent)
	if gameProcessedEvent then return end
	if input.KeyCode == SET_WAYPOINT_KEY then
		setWaypoint()
	elseif input.KeyCode == TOGGLE_KEY then
		toggleLock()
	end
end)

RunService.Heartbeat:Connect(function()
	if locked and waypointCFrame then
		local hrp = getHRP()
		if hrp then
			hrp.CFrame = waypointCFrame
		end
	end
end)
